- 通用函数装饰器

```python
def set_func(func):
    def call_func(*args, **kwargs):
        print("装饰1")
        ret = func(*args, **kwargs)
        print("装饰2")
        return ret
    return call_func

@set_func
def test(num, *args, **kwargs):
    print("-----", num)
    print("-----", args)
    print("-----", kwargs)
    return 'ok'

ret = test(100, 200, 300, mm=150)
print(ret)
```

- 通用类装饰器

```python
class Test(object):
    def __init__(self, func):
        self.func = func

    def __call__(self, *args, **kwargs):
        print("这里添加装饰功能")
        ret = self.func(*args, **kwargs)
        print("这里添加装饰功能")
        return  ret

@Test  #test = Test(test)
def test(*args, **kwargs):
    print(args)
    print(kwargs)
    return 'haha'

print(test(10, 200, mm=100))
```

- 多个装饰器装饰一个函数

```python
def set_func1(func):
    def call_func(*args, **kwargs):
        print("装饰1")
        ret = func(*args, **kwargs)
        print("装饰2")
        return ret
    return call_func

def set_func2(func):
    def call_func(*args, **kwargs):
        print("装饰3")
        ret = func(*args, **kwargs)
        print("装饰4")
        return ret
    return call_func

@set_func1
@set_func2
def test(num, *args, **kwargs):
    print("-----", num)
    print("-----", args)
    print("-----", kwargs)
    return 'ok'

ret = test(100, 200, 300, mm=150)
print(ret)

'''
结果：
装饰1
装饰3
----- 100
----- (200, 300)
----- {'mm': 150}
装饰4
装饰2
ok
'''

```

- 带参数的装饰器

当装饰器带上参数， 就会分为两部：
1. 调用set_func并且将1当作实参传递
2. 用上一步调用的返回值 当做 装饰器对test1函数进行装饰

```python
def set_level(level_num):
    def set_func(func):
        def call_func(*args, **kwargs):
            if level_num == 1:
                print("level 1")
            elif level_num == 2:
                print("level 2")
            print("装饰1")
            ret = func(*args, **kwargs)
            print("装饰2")
            return ret
        return call_func
    return set_func

@set_level(1)
def test1(num, *args, **kwargs):
    print("-----", num)
    print("-----", args)
    print("-----", kwargs)
    return 'ok'

@set_level(2)
def test2(num, *args, **kwargs):
    print("-----", num)
    print("-----", args)
    print("-----", kwargs)
    return 'ok2'

ret = test1(100, 200, 300, mm=150)
print(ret)
ret = test2(100, 200, 300, mm=150)
print(ret)

# 结果
'''
level 1
装饰1
----- 100
----- (200, 300)
----- {'mm': 150}
装饰2
ok
level 2
装饰1
----- 100
----- (200, 300)
----- {'mm': 150}
装饰2
ok2
'''

```

- 使用带参数的装饰器构建mini web框架
```python
import re

# 全局变量用大写
URL_FUNC_DICT = dict()

# 装饰器
def route(url):
    def set_func(func):
        URL_FUNC_DICT[url] = func
        def call_func(*args, **kwargs):
            return func(*args, **kwargs)
        return call_func
    return set_func

@route("index.py")
def index():
    with open("./templates/index.html") as f:
        content = f.read()
    my_stock_info = "你好"
    content = re.sub(r"\{%content%\}", my_stock_info, content)
    return content

@route("/center.py")
def center():
    with open("./templates/index.html") as f:
        content = f.read()
    my_stock_info = "这是从mysql查询出来的数据。。。"
    content = re.sub(r"\{%content%\}", my_stock_info, content)
    return content


def application(env, start_response):
    start_response('200 OK', [('Content-Type', 'text/html;charset=utf-8')])
    file_name = env['PATH_INFO']
    func = URL_FUNC_DICT(file_name)
    return func()
```
