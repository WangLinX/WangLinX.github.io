- 通用函数装饰器

```python
def set_func(func):
    def call_func(*args, **kwargs):
        print("装饰1")
        ret = func(*args, **kwargs)
        print("装饰2")
        return ret
    return call_func

@set_func
def test(num, *args, **kwargs):
    print("-----", num)
    print("-----", args)
    print("-----", kwargs)
    return 'ok'

ret = test(100, 200, 300, mm=150)
print(ret)
```

- 通用类装饰器

```python
class Test(object):
    def __init__(self, func):
        self.func = func

    def __call__(self, *args, **kwargs):
        print("这里添加装饰功能")
        ret = self.func(*args, **kwargs)
        print("这里添加装饰功能")
        return  ret

@Test  #test = Test(test)
def test(*args, **kwargs):
    print(args)
    print(kwargs)
    return 'haha'

print(test(10, 200, mm=100))
```

- 多个装饰器装饰一个函数

```python
def set_func1(func):
    def call_func(*args, **kwargs):
        print("装饰1")
        ret = func(*args, **kwargs)
        print("装饰2")
        return ret
    return call_func

def set_func2(func):
    def call_func(*args, **kwargs):
        print("装饰3")
        ret = func(*args, **kwargs)
        print("装饰4")
        return ret
    return call_func

@set_func1
@set_func2
def test(num, *args, **kwargs):
    print("-----", num)
    print("-----", args)
    print("-----", kwargs)
    return 'ok'

ret = test(100, 200, 300, mm=150)
print(ret)

'''
结果：
装饰1
装饰3
----- 100
----- (200, 300)
----- {'mm': 150}
装饰4
装饰2
ok
'''
```
